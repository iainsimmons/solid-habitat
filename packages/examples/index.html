Email Subscribe with store

Use a child <script type="text/props"> tag with valid JSON for passing props to components

<div data-habitat="email-subscribe" class="preview">
  <!-- path: src/components/email-subscribe/index.tsx -->
  <script type="text/props">
    {
      "showName": false,
      "showEmail": true
    }
  </script>
</div>

Use data-b-prop for boolean props

<div data-habitat="email-subscribe" class="preview" data-b-prop-show-name="true" data-b-prop-show-email="false"></div>
<!-- path: src/components/email-subscribe/index.tsx -->
